cmake_minimum_required(VERSION 3.15)

set(BGFX_ROOT "${CMAKE_CURRENT_LIST_DIR}/..")

set(BGFX_INCLUDE_DIR "${BGFX_ROOT}/include")
set(BGFX_LIB_DIR "${BGFX_ROOT}/lib")

add_library(bgfx::bgfx STATIC IMPORTED)
set_target_properties(bgfx::bgfx PROPERTIES
    IMPORTED_LOCATION "${BGFX_LIB_DIR}/bgfxRelease.lib"
    INTERFACE_INCLUDE_DIRECTORIES "${BGFX_INCLUDE_DIR}"
)

add_library(bgfx::bx STATIC IMPORTED)
set_target_properties(bgfx::bx PROPERTIES
    IMPORTED_LOCATION "${BGFX_LIB_DIR}/bxRelease.lib"
    INTERFACE_INCLUDE_DIRECTORIES "${BGFX_INCLUDE_DIR}"
)

add_library(bgfx::bimg STATIC IMPORTED)
set_target_properties(bgfx::bimg PROPERTIES
    IMPORTED_LOCATION "${BGFX_LIB_DIR}/bimgRelease.lib"
    INTERFACE_INCLUDE_DIRECTORIES "${BGFX_INCLUDE_DIR}"
)

add_library(bgfx::bimg_decode STATIC IMPORTED)
set_target_properties(bgfx::bimg_decode PROPERTIES
    IMPORTED_LOCATION "${BGFX_LIB_DIR}/bimg_decodeRelease.lib"
    INTERFACE_INCLUDE_DIRECTORIES "${BGFX_INCLUDE_DIR}"
)
